  <feature>
    <name>normalizeObservation</name>
    <files>src/ai/vecnormalize.ts, tests/ai/vecnormalize.test.ts</files>
    <behavior>
      Applies the SB3 VecNormalize formula exactly: clip((obs[i] - mean[i]) / sqrt(var[i] + epsilon), -clip_obs, clip_obs)

      Cases:
      - Zero mean, unit var, no clipping → obs returned unchanged (within float precision)
      - Value well above clip_obs → clipped to clip_obs
      - Value well below -clip_obs → clipped to -clip_obs
      - Different mean/var per element → each normalized independently
      - epsilon=1e-8, clip_obs=10.0 defaults match SB3 defaults
      - Output array has same length as input obs array (14 elements for full obs vector)
    </behavior>
    <implementation>
      Create src/ai/vecnormalize.ts:
      ```typescript
      export interface VecNormStats {
        obs_mean: number[];
        obs_var: number[];
        clip_obs: number;
        epsilon: number;
      }

      export function normalizeObservation(obs: number[], stats: VecNormStats): number[] {
        const { obs_mean, obs_var, clip_obs, epsilon } = stats;
        return obs.map((v, i) => {
          const normalized = (v - obs_mean[i]) / Math.sqrt(obs_var[i] + epsilon);
          return Math.max(-clip_obs, Math.min(clip_obs, normalized));
        });
      }
      ```
    </implementation>
  </feature>
